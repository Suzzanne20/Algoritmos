#include "iostream"  // Biblioteca estandar que permite entradas y salidas
#include "string"

using namespace std;

int main()
{
    
    cout << "Suma de 2 numeros" << "\n";  //Mensaje de pantalla

    
    float numero1, numero2;    //declaración de variables tipo decimales

     
    cout << "Por favor ingrese el primer valor: " << "\n"; //   ->  \n = salto de linea  

    cin >> numero1;    // se captura el valor ingresado en la variable


    cout << "Por favor ingrese el segundo valor: " << "\n";  //Mensaje de pantalla

    cin >> numero2;   // se captura el valor ingresado en la variable

  
    cout << numero1 << " + " << numero2 << " = " << numero1 + numero2;    // se muestran los valores y se realiza la operación muestrando en pantalla

   return 0;
}
